\begin{frame}{Le contenu de la modélisation}
	\begin{itemize}
		\item<1-> Slimuler le développement résidentiel à différents niveaux d'échelles et selon plusieurs contraintes
		\item<2-> Exprimer divers critères d'évaluation de la plannification~:
		\begin{itemize}	
			\item<3-> La création de logements
			\item<4-> La densité de logements par hectare
			\item<5-> L'accessibilité (aux transports en communs, ...)
		\end{itemize}
		\item<6-> Pas d'évolution temporelles
		\item<6-> Pas de reproduction des phénomènes anterieurs
		\item<7-> Solutions libres et open-source pour une recherche \textbf{vérifiable} et \textbf{reproductible}
	\end{itemize}
\end{frame}

\begin{frame}{État de l'art des outils d'aide à la décision dans l'aménagement à visée prospectif}
	\only<1>{Modèle d'\textbf{aide à la décision multi-critères} (ref) permet de~: %TODO ref
	\begin{itemize}
		\item Prends en compte de nombreux paramètres pondérés pour dégager une décision
		\item Pondérer les différents systèmes intéressants pour le développement résidentiel
	\end{itemize}
	}
	\only<2>{Problèmes des modèles d'\textbf{aide à la décision multi-critères}~:
	\begin{itemize}
		\item Solutions pré-déterminées
		\item Un phénomène plutôt qu'une interaction complexe entre différents systèmes
	\end{itemize}
	}
	\only<3>{
	Les modèles d'\textbf{étude du marché immobilier} (UrbanSimul (), ...) %TODO ref
	\begin{itemize}
		\item Analyser le marché de l'immobilier, la pression foncière
		\item Propose des prédictions sur l'état du marché et sur les futures urbanisations
	\end{itemize}
	}
	\only<4>{
	Problèmes des \textbf{modèles d'étude du marché immobilier} (UrbanSimul (), ...) ~:
	\begin{itemize}
			\item Peu adapté à représenter la compléxité des mécanismes multi-échelles d'un développement résidentiel 
			\item Résultats dépendants principalement du marché et non des objectifs de la planification
%			\item Très contextualisé
	\end{itemize}}
	\only<5>{Ces outils sont appliqués à la résolution d'\textbf{un} problème défini et \textbf{peu interopérable}}
\end{frame}

\begin{frame}{État de l'art des modèles de simulation du développement résidentiel de faits stylisés}
	\only<1>{Fait stylisé~: modélisation de phénomènes complexes grâce à des objets géométriques \textit{simplifiés}\\}
	\only<2>{Les \textbf{automates cellulaires} (Yeh 2015, Mustafa 2018) sont utilisés pour 
		\begin{itemize}
			\item fournir une représentation synthétique et simplifiée de l'espace grâce à un carroyage
			\item simuler des changements d'états dynamiques pour chacune des cellules en fonction d'équations 
		\end{itemize}
	}
	\only<3>{
	Les \textbf{automates cellulaires} (Yeh 2015, Mustafa 2018) ne permettent pas de 
		\begin{itemize}
			\item traiter des formes géographiques complexes (parcelles, bâtiments)
		\end{itemize}
	}
	\only<4>{
	Les \textbf{modèles Multi-Agents} (Artznete 10, ...) permettent de 
	\begin{itemize}
		\item représenter un système en reproduisant les comportements d'agents et leurs interactions 
		\item permet l'émergence de configurations particulières
	\end{itemize}
	}
	\only<5>{
		Problèmes des \textbf{modèles Multi-Agents} (Artznete 10, ...)~:
		\begin{itemize}
			\item Très dynamique
			\item Modélisation de notre problème trop compliquée %(définition des agents, des interactions et des contraintes)
		\end{itemize}
	}
	\only<6>{Les modèles simplifiés ne sont pas assez descriptifs pour représenter le développement résidentiel à un niveau très local}
\end{frame}

\begin{frame}{État de l'art des modèles de simulation intégré de développement résidentiel}
	\only<1>{
		Les approches intégrées, tel que les LUTI (Land-Use and Transportation Interaction), permettent de~:
		\begin{itemize}
			\item Simuler les interactions entre différents modèles (occupation du sol, transport, systèmes économiques ...)
			\item Articule différents systèmes modélisés
			\item Approche prospectives %TODO redire
		\end{itemize}
	}
	\only<2->{
		Problèmes avec les approches intégrés, tel que les LUTI~:
		\begin{itemize}
			\item mise en œuvre conséquente
			\item modélistation des mobilités non nécessaire à notre problème
		\end{itemize}
		\uncover<3>{Approche intéressante du \textbf{couplage} de \textbf{modules} agissant à \textbf{différentes échelles}}
	}
\end{frame}

			\end{itemize}
		\end{block}
	}
	\only<3>{
		\begin{block}{Scénario d'extension résidentielle ciblée}
			\begin{itemize}
				\item Potentiellement variables sur de très grandes zones face au \textbf{caractère aléatoire}
				\item Très variable selon la taille des cellules
				\item variable selon la position éloignée de la grille de décomposition
			\end{itemize}
		\end{block}	
	}
	\only<4>{
		\begin{block}{Scénario d'intensification de la densitée résidentielle}
			\begin{itemize}

				\item localement très sensible à la position de la grille de décomposition
			\end{itemize}
		\end{block}	
	}
\end{frame}

\begin{frame}{Documents de planification locale - Les PLU}
Le \textbf{Plan Local de l’Urbanisme (PLU)} détaille et spatialise les contraintes de constructibilité au sein d’une commune
\begin{itemize}
\item a des \textbf{effets directs sur la constructibilité} mais ne planifie pas la construction 
\item \textbf{donne un cadre} pour la création de programmes de construction de logements \textit{(OAP, ZAC, ZAD)}
\item se compose en partie d'un \textbf{zonage} et d'un \textbf{règlement}
\end{itemize}
\end{frame}
