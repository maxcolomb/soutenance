\begin{frame}{État de l'art des modèles de développement résidentiel}
	\only<1>{Les modèles d'\textbf{occupation du sol} (Batty 97, Arentze 08, Mustapha 18). %todo ajouter images?! peut être en mettre un tout petit peu plus
		\begin{itemize}
			\item Simulation du changement d'état de l'occupation du sol et particulièrement de l'urbanisation
			\item Poursuis de nombreux objectifs~:
			\begin{itemize}
				\item Exploratoire (Potentiels d'urbanisation, )
				\item Prévisionnel (marché immobilier (), changements démographiques (Destinie ()) %todo refs

		\end{itemize}
		\only<1>{Inconvénients~: 
		\begin{itemize}
			\item Espace géographique simplifié
			\item Peu de place pour la prise en compte des réglementations
			\item Souvent calibrés sur des dynamiques passées
		\end{itemize}}
	}

\begin{frame}{Documents de planification locale - Les PLU}
Le \textbf{Plan Local de l’Urbanisme (PLU)} détaille et spatialise les contraintes de constructibilité au sein d’une commune
\begin{itemize}
\item a des \textbf{effets directs sur la constructibilité} mais ne planifie pas la construction 
\item \textbf{donne un cadre} pour la création de programmes de construction de logements \textit{(OAP, ZAC, ZAD)}
\item se compose en partie d'un \textbf{zonage} et d'un \textbf{règlement}
\end{itemize}
\end{frame}
